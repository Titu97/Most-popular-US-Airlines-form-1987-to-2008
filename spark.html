import pandas
from collections import Counter
wee=[]
for i in range(22):
    ty= t.filter(t.Year == 1987+i)    
    ty1=ty.groupby('UniqueCarrier').count() 
    a=ty1.sort("count", ascending=False).take(10)
    
    we=[]
    for k in range(10):
        s=[a[k][0]]*a[k][1]
        we.extend(s)
    wee.append(we)
    letter_counts = Counter(we)
    df = pandas.DataFrame.from_dict(letter_counts, orient='index')
    df.plot(title=str(1987+i),kind='bar')
